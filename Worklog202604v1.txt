    <script>
        /* =========================================
           狀態管理與核心邏輯 (State Management)

        // 核心資料結構
        let appState = {
            focusHistory: [],
            checklist: [],
            notes: "",
            workProgress: [] // 新增工作進度狀態
        };

        // 追蹤是否有未儲存(未匯出)的變更
        let isDirty = false;

        // 標記資料已變更，觸發防呆機制
        function markDirty() {
            isDirty = true;
            document.getElementById('unsaved-dot').classList.remove('hidden');
        }

        // 標記資料已乾淨(已匯出/剛匯入)
        function markClean() {
            isDirty = false;
            document.getElementById('unsaved-dot').classList.add('hidden');
        }

        // 防呆機制：關閉或重整前警告
        window.addEventListener('beforeunload', (e) => {
            if (isDirty) {
                const msg = "You have unsaved changes. Are you sure you want to leave?";
                e.returnValue = msg; // 標準寫法
                return msg;
            }
        });

        /* =========================================
           UI 初始化與動態主題色彩生成
           ========================================= */

        function initApp() {
            // 1. 設定日期顯示
            const now = new Date();
            const dateOptions = { year: 'numeric', month: 'long', day: 'numeric', weekday: 'long' };
            document.getElementById('header-date').textContent = now.toLocaleDateString('en-US', dateOptions);

            // 2. 預設「預計完成時間」為當前時間的 1 小時後
            now.setHours(now.getHours() + 1);
            const hours = String(now.getHours()).padStart(2, '0');
            const minutes = String(now.getMinutes()).padStart(2, '0');
            document.getElementById('focus-time').value = `${hours}:${minutes}`;

            // 3. 根據日期生成動態主題色 (大膽且和諧)
            generateDailyTheme();

            // 4. 綁定事件監聽
            setupDragAndDrop();
            document.getElementById('import-file').addEventListener('change', handleFileUpload);
            
            // 初始渲染一次空的狀態
            renderAll();
        }

        // 根據當前日期算出一個 Hash，並轉為 HSL 顏色
        function generateDailyTheme() {
            const dateStr = new Date().toISOString().split('T')[0]; // "YYYY-MM-DD"
            let hash = 0;
            for (let i = 0; i < dateStr.length; i++) {
                hash = dateStr.charCodeAt(i) + ((hash << 5) - hash);
            }
            // 產生 0-359 之間的主色調 (Hue)
            const huePrimary = Math.abs(hash) % 360;
            // 產生對比或相鄰的副色調
            const hueSecondary = (huePrimary + 45) % 360; 

            // 設定 CSS 變數，大膽色彩：飽和度 85%，亮度 55%
            document.documentElement.style.setProperty('--color-primary', `hsl(${huePrimary}, 85%, 55%)`);
            document.documentElement.style.setProperty('--color-secondary', `hsl(${hueSecondary}, 85%, 65%)`);
            
            // 給背景一個極淡的主題色，增加整體沉浸感
            document.getElementById('body-bg').style.backgroundColor = `hsl(${huePrimary}, 30%, 98%)`;
        }

        /* =========================================
           JSON 檔案匯出/匯入邏輯 (重要！)
           ========================================= */

        /**
         * [匯出邏輯]
         * 1. 將 `appState` 物件轉換為 JSON 字串。
         * 2. 使用 Blob (Binary Large Object) 將字串封裝成一個檔案物件。
         * 3. 建立一個暫時的 URL (URL.createObjectURL)。
         * 4. 創造一個隱藏的 <a> 標籤，設定 download 屬性及檔案名稱。
         * 5. 觸發點擊下載，並將 isDirty 設為 false。
         */
        function exportData() {
            // 更新 notes 狀態
            appState.notes = document.getElementById('daily-notes').value;

            const dataStr = JSON.stringify(appState, null, 2);
            const blob = new Blob([dataStr], { type: "application/json" });
            const url = URL.createObjectURL(blob);
            
            const dateStr = new Date().toISOString().split('T')[0].replace(/-/g, '');
            const filename = `work-log-${dateStr}.json`;

            const a = document.createElement('a');
            a.href = url;
            a.download = filename;
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
            URL.revokeObjectURL(url);

            markClean(); // 下載完畢，解除警告狀態
            alert("? Data exported successfully!");
        }

        /**
         * [匯入邏輯]
         * 1. 取得使用者上傳或拖曳的 File 物件。
         * 2. 使用 FileReader 讀取檔案內容為純文字 (readAsText)。
         * 3. 透過 JSON.parse 解析字串，還原成 JavaScript 物件。
         * 4. 將資料灌回 `appState` 並重新渲染畫面。
         */
        function handleFileUpload(event) {
            const file = event.target.files[0];
            if (!file) return;
            processFile(file);
            event.target.value = ''; // 清空 input 以便重複選擇同檔
        }

        function processFile(file) {
            if (file.type !== "application/json" && !file.name.endsWith(".json")) {
                alert("? Please upload a valid JSON file!");
                return;
            }

            const reader = new FileReader();
            reader.onload = function(e) {
                try {
                    const loadedData = JSON.parse(e.target.result);
                    // 簡單驗證資料格式
                    if (loadedData.focusHistory !== undefined && loadedData.checklist !== undefined) {
                        // 確保舊版 JSON 載入時也有 workProgress 屬性
                        if (!loadedData.workProgress) loadedData.workProgress = [];
                        
                        appState = loadedData;
                        renderAll();
                        markClean(); // 剛載入的資料視為乾淨狀態
                    } else {
                        alert("? Invalid JSON format for this system. Load failed.");
                    }
                } catch (err) {
                    alert("? JSON parsing failed. The file might be corrupted.");
                }
            };
            reader.readAsText(file);
        }

        /* 拖曳載入 (Drag & Drop) 視覺與邏輯處理 */
        function setupDragAndDrop() {
            const overlay = document.getElementById('dropzone-overlay');

            window.addEventListener('dragover', (e) => {
                e.preventDefault();
                overlay.classList.add('active');
            });

            window.addEventListener('dragleave', (e) => {
                // 確保滑鼠真的離開視窗才隱藏
                if (e.clientX === 0 || e.clientY === 0) {
                    overlay.classList.remove('active');
                }
            });

            window.addEventListener('drop', (e) => {
                e.preventDefault();
                overlay.classList.remove('active');
                
                if (e.dataTransfer.files.length > 0) {
                    processFile(e.dataTransfer.files[0]);
                }
            });
        }

        // 統一重新渲染所有資料到畫面上
        function renderAll() {
            renderChecklist();
            renderFocusHistory();
            renderWorkProgress(); // 渲染工作進度
            document.getElementById('daily-notes').value = appState.notes || "";
        }

        /* =========================================
           區塊 1: Focus Hour & 番茄鐘邏輯
           ========================================= */

        let timerInterval = null;
        let timeLeft = 25 * 60; // 預設 25 分鐘 (秒)
        let isTimerRunning = false;
        let isWorkMode = true; // true = 25分工作, false = 5分休息

        function startFocus() {
            const topic = document.getElementById('focus-topic').value.trim();
            if (!topic) {
                alert("Please enter a focus topic first!");
                document.getElementById('focus-topic').focus();
                return;
            }

            // 顯示番茄鐘 UI，隱藏開始按鈕
            document.getElementById('pomodoro-container').classList.remove('hidden');
            document.getElementById('start-focus-btn').classList.add('hidden');
            
            // 確保重置狀態並自動開始
            resetTimer(false);
            toggleTimer();
            markDirty();
        }

        function toggleTimer() {
            const btn = document.getElementById('timer-pause-btn');
            if (isTimerRunning) {
                clearInterval(timerInterval);
                isTimerRunning = false;
                btn.textContent = "Resume";
                btn.classList.replace('bg-yellow-500', 'bg-green-500');
                btn.classList.replace('hover:bg-yellow-600', 'hover:bg-green-600');
            } else {
                timerInterval = setInterval(updateTimerDisplay, 1000);
                isTimerRunning = true;
                btn.textContent = "Pause";
                btn.classList.replace('bg-green-500', 'bg-yellow-500');
                btn.classList.replace('hover:bg-green-600', 'hover:bg-yellow-600');
            }
        }

        function resetTimer(manual = true) {
            clearInterval(timerInterval);
            isTimerRunning = false;
            isWorkMode = true;
            timeLeft = 25 * 60;
            updateStatusText();
            formatAndShowTime();
            
            const btn = document.getElementById('timer-pause-btn');
            btn.textContent = "Start/Pause";
            btn.classList.replace('bg-green-500', 'bg-yellow-500');
            btn.classList.replace('hover:bg-green-600', 'hover:bg-yellow-600');

            if (manual) markDirty();
        }

        function updateTimerDisplay() {
            if (timeLeft > 0) {
                timeLeft--;
                formatAndShowTime();
            } else {
                // 計時結束
                clearInterval(timerInterval);
                isTimerRunning = false;
                playAlarm();

                if (isWorkMode) {
                    // 工作結束，切換到休息
                    alert("?? 25 minutes focus finished! Take a break.");
                    isWorkMode = false;
                    timeLeft = 5 * 60;
                    
                    // 顯示填寫紀錄的區塊
                    document.getElementById('log-entry-container').classList.remove('hidden');
                    document.getElementById('log-entry-container').classList.add('flex');
                    document.getElementById('focus-log-input').focus();
                } else {
                    // 休息結束
                    alert("? Break finished! Ready for the next focus.");
                    isWorkMode = true;
                    timeLeft = 25 * 60;
                }
                updateStatusText();
                formatAndShowTime();
                
                const btn = document.getElementById('timer-pause-btn');
                btn.textContent = "Resume";
                btn.classList.replace('bg-yellow-500', 'bg-green-500');
            }
        }

        function formatAndShowTime() {
            const m = String(Math.floor(timeLeft / 60)).padStart(2, '0');
            const s = String(timeLeft % 60).padStart(2, '0');
            document.getElementById('timer-display').textContent = `${m}:${s}`;
        }

        function playAlarm() {
            // 簡單的音效提示
            const audioCtx = new (window.AudioContext || window.webkitAudioContext)();
            const oscillator = audioCtx.createOscillator();
            oscillator.type = 'sine';
            oscillator.frequency.setValueAtTime(880, audioCtx.currentTime); // A5
            oscillator.connect(audioCtx.destination);
            oscillator.start();
            oscillator.stop(audioCtx.currentTime + 0.5);
        }

        // 儲存工作紀錄
        function saveFocusLog() {
            const topic = document.getElementById('focus-topic').value;
            const targetTime = document.getElementById('focus-time').value;
            const logContent = document.getElementById('focus-log-input').value.trim();

            if (!logContent) {
                alert("Please fill in the completed content!");
                return;
            }

            const now = new Date();
            const timeStr = `${String(now.getHours()).padStart(2, '0')}:${String(now.getMinutes()).padStart(2, '0')}`;

            appState.focusHistory.push({
                topic: topic || "Unnamed Task",
                plannedTime: targetTime,
                actualTime: timeStr,
                log: logContent
            });

            markDirty();
            renderFocusHistory();

            // 隱藏紀錄填寫區，恢復原始狀態，準備下次專注
            document.getElementById('log-entry-container').classList.add('hidden');
            document.getElementById('log-entry-container').classList.remove('flex');
            document.getElementById('focus-log-input').value = "";
            document.getElementById('focus-topic').value = "";
            
            document.getElementById('pomodoro-container').classList.add('hidden');
            document.getElementById('start-focus-btn').classList.remove('hidden');
            resetTimer(false);
        }

        function renderFocusHistory() {
            const list = document.getElementById('focus-history-list');
            list.innerHTML = "";

            if (appState.focusHistory.length === 0) {
                list.innerHTML = `<li class="text-sm text-gray-400 italic text-center py-4">No records yet. Start focusing today!</li>`;
                return;
            }

            // 倒序顯示，最新在最上
            [...appState.focusHistory].reverse().forEach(record => {
                const li = document.createElement('li');
                li.className = "p-3 bg-white rounded-lg border border-gray-100 shadow-sm flex flex-col gap-1";
                li.innerHTML = `
                    <div class="flex justify-between items-center">
                        <span class="font-bold text-gray-700 text-sm">${record.topic}</span>
                        <span class="text-xs text-gray-400"><i class="ph ph-clock"></i> Completed at ${record.actualTime}</span>
                    </div>
                    <p class="text-sm text-gray-600 mt-1">${record.log}</p>
                `;
                list.appendChild(li);
            });
        }

           區塊 2: Checklist 邏輯
           ========================================= */

        function addChecklistItem(e) {
            e.preventDefault();
            const input = document.getElementById('checklist-input');
            const text = input.value.trim();
            if (!text) return;

            appState.checklist.push({
                id: Date.now().toString(),
                text: text,
                completed: false
            });

            input.value = "";
            markDirty();
            renderChecklist();
        }

        function toggleChecklist(id) {
            const item = appState.checklist.find(i => i.id === id);
            if (item) {
                item.completed = !item.completed;
                markDirty();
                renderChecklist();
            }
        }

        function deleteChecklistItem(id) {
            appState.checklist = appState.checklist.filter(i => i.id !== id);
            markDirty();
            renderChecklist();
        }

        function clearCompletedChecklist() {
            const oldLength = appState.checklist.length;
            appState.checklist = appState.checklist.filter(i => !i.completed);
            
            if (appState.checklist.length !== oldLength) {
                markDirty();
                renderChecklist();
            }
        }

        function renderChecklist() {
            const container = document.getElementById('checklist-container');
            container.innerHTML = "";

            if (appState.checklist.length === 0) {
                container.innerHTML = `<div class="text-sm text-gray-400 italic text-center py-4">No pending tasks</div>`;
                return;
            }

            appState.checklist.forEach(item => {
                const li = document.createElement('li');
                li.className = `flex items-center justify-between p-3 rounded-lg border border-gray-100 bg-white shadow-sm hover:border-[var(--color-primary)] transition-colors group`;
                
                li.innerHTML = `
                    <div class="flex items-center gap-3 flex-1 overflow-hidden" onclick="toggleChecklist('${item.id}')" style="cursor: pointer;">
                        <div class="w-5 h-5 rounded border ${item.completed ? 'theme-bg theme-border text-white flex items-center justify-center' : 'border-gray-300'} transition-colors">
                            ${item.completed ? '<i class="ph ph-check text-xs"></i>' : ''}
                        </div>
                        <span class="truncate ${item.completed ? 'completed-item' : 'text-gray-700 font-medium'}">${item.text}</span>
                    </div>
                    <button onclick="deleteChecklistItem('${item.id}')" class="text-gray-300 hover:text-red-500 opacity-0 group-hover:opacity-100 transition-opacity p-1">
                        <i class="ph ph-x"></i>
                    </button>
                `;
                container.appendChild(li);
            });
        }

        /* =========================================
           區塊 4: Work Progress 邏輯
           ========================================= */
        
        let editingProgressId = null; // 紀錄目前正在編輯的進度 ID

        function handleWorkProgressSubmit(e) {
            e.preventDefault();
            const job = document.getElementById('wp-job').value.trim();
            const nature = document.getElementById('wp-nature').value.trim();
            const solo = document.getElementById('wp-solo').value.trim();
            const content = document.getElementById('wp-content').value.trim();

            if (!job && !nature && !solo && !content) return; // 避免送出全空

            if (editingProgressId) {
                // 更新現有資料
                const item = appState.workProgress.find(i => i.id === editingProgressId);
                if (item) {
                    item.jobNumber = job;
                    item.nature = nature;
                    item.solo = solo;
                    item.content = content;
                }
                editingProgressId = null;
                const btn = document.getElementById('wp-submit-btn');
                btn.innerHTML = `<i class="ph ph-plus-circle mr-1"></i> Add Progress`;
            } else {
                // 建立新資料
                appState.workProgress.push({
                    id: Date.now().toString(),
                    jobNumber: job,
                    nature: nature,
                    solo: solo,
                    content: content
                });
            }

            e.target.reset(); // 清空表單
            markDirty();
            renderWorkProgress();
        }

        function editWorkProgress(id) {
            const item = appState.workProgress.find(i => i.id === id);
            if (!item) return;

            // 將資料填回表單
            document.getElementById('wp-job').value = item.jobNumber;
            document.getElementById('wp-nature').value = item.nature;
            document.getElementById('wp-solo').value = item.solo;
            document.getElementById('wp-content').value = item.content;

            editingProgressId = id;
            
            // 變更按鈕外觀與文字
            const btn = document.getElementById('wp-submit-btn');
            btn.innerHTML = `<i class="ph ph-floppy-disk mr-1"></i> Save Changes`;

            // 滑動回表單區塊方便編輯
            document.getElementById('work-progress-section').scrollIntoView({ behavior: 'smooth' });
        }

        function deleteWorkProgress(id) {
            if (confirm("Are you sure you want to delete this progress?")) {
                appState.workProgress = appState.workProgress.filter(i => i.id !== id);
                
                // 如果正在編輯被刪除的項目，則重置表單
                if (editingProgressId === id) {
                    editingProgressId = null;
                    document.getElementById('wp-job').form.reset();
                    document.getElementById('wp-submit-btn').innerHTML = `<i class="ph ph-plus-circle mr-1"></i> Add Progress`;
                }

                markDirty();
                renderWorkProgress();
            }
        }

        function renderWorkProgress() {
            const list = document.getElementById('wp-list');
            list.innerHTML = "";

            if (!appState.workProgress || appState.workProgress.length === 0) {
                list.innerHTML = `<div class="col-span-full text-sm text-gray-400 italic text-center py-6">No work progress created yet</div>`;
                return;
            }

            // 反轉陣列以最新的排在最前面
            [...appState.workProgress].reverse().forEach(item => {
                const card = document.createElement('div');
                card.className = "bg-white p-5 rounded-xl border border-gray-200 shadow-sm flex flex-col gap-2 hover:shadow-md transition-shadow";
                
                // 將換行符號轉為 <br> 以便在 HTML 正確顯示
                const formattedContent = item.content.replace(/\n/g, '<br>');

                card.innerHTML = `
                    <div class="flex flex-wrap items-center gap-2 mb-1 border-b pb-2">
                        ${item.jobNumber ? `<span class="px-2 py-1 bg-gray-100 text-gray-700 text-xs font-bold rounded">#${item.jobNumber}</span>` : ''}
                        ${item.nature ? `<span class="px-2 py-1 bg-blue-50 text-blue-600 border border-blue-100 text-xs font-semibold rounded">${item.nature}</span>` : ''}
                        ${item.solo ? `<span class="px-2 py-1 bg-purple-50 text-purple-600 border border-purple-100 text-xs font-semibold rounded">SOLO: ${item.solo}</span>` : ''}
                        <div class="ml-auto flex gap-2">
                            <button onclick="editWorkProgress('${item.id}')" title="編輯" class="text-gray-400 hover:text-blue-500 transition-colors">
                                <i class="ph ph-pencil-simple text-lg"></i>
                            </button>
                            <button onclick="deleteWorkProgress('${item.id}')" title="刪除" class="text-gray-400 hover:text-red-500 transition-colors">
                                <i class="ph ph-trash text-lg"></i>
                            </button>
                        </div>
                    </div>
                    <div class="text-sm text-gray-700 leading-relaxed mt-1 break-words">
                        ${formattedContent || '<span class="text-gray-400 italic">No details provided</span>'}
                    </div>
                `;
                list.appendChild(card);
            });
        }
